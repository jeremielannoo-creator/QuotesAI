    <script>
        document.addEventListener('DOMContentLoaded', function() {
            // Références aux éléments DOM
            const pexelsApiInput = document.getElementById('pexels-api');
            const pixabayApiInput = document.getElementById('pixabay-api');
            const saveApiButton = document.getElementById('save-api-keys');
            const generateQuoteButton = document.getElementById('generate-quote');
            const findVideoButton = document.getElementById('find-video');
            const quoteResult = document.getElementById('quote-result');
            const quoteText = document.getElementById('quote-text');
            const quoteAuthor = document.getElementById('quote-author');
            const quoteTags = document.getElementById('quote-tags');
            const loading = document.getElementById('loading');
            const previewContent = document.getElementById('preview-content');
            const noPreview = document.getElementById('no-preview');
            const quoteImage = document.getElementById('quote-image');
            const videoPreview = document.getElementById('video-preview');
            const videoElement = document.getElementById('video-element');
            const instagramBtn = document.getElementById('instagram-btn');
            const tiktokBtn = document.getElementById('tiktok-btn');
            const apiSavedMessage = document.getElementById('api-saved-message');
            
            // Variables pour stocker les données
            let currentQuote = null;
            let currentVideo = null;
            let apiKeys = {
                pexels: '',
                pixabay: ''
            };
            
            // Charger les clés API sauvegardées
            const savedApiKeys = localStorage.getItem('citationAppApiKeys');
            if (savedApiKeys) {
                try {
                    apiKeys = JSON.parse(savedApiKeys);
                    pexelsApiInput.value = apiKeys.pexels;
                    pixabayApiInput.value = apiKeys.pixabay;
                } catch (e) {
                    console.error('Erreur lors du chargement des clés API:', e);
                }
            }
            
            // Enregistrer les clés API
            saveApiButton.addEventListener('click', function() {
                apiKeys.pexels = pexelsApiInput.value.trim();
                apiKeys.pixabay = pixabayApiInput.value.trim();
                
                localStorage.setItem('citationAppApiKeys', JSON.stringify(apiKeys));
                
                // Afficher le message de confirmation
                apiSavedMessage.classList.remove('hidden');
                setTimeout(() => {
                    apiSavedMessage.classList.add('hidden');
                }, 3000);
            });
            
            // Générer une citation aléatoire
            generateQuoteButton.addEventListener('click', async function() {
                try {
                    loading.classList.remove('hidden');
                    
                    const response = await fetch('https://api.quotable.io/random');
                    if (!response.ok) {
                        throw new Error('Erreur lors de la récupération de la citation');
                    }
                    
                    const quoteData = await response.json();
                    currentQuote = quoteData;
                    
                    // Afficher la citation
                    quoteText.textContent = `"${quoteData.content}"`;
                    quoteAuthor.textContent = `- ${quoteData.author}`;
                    
                    // Afficher les tags
                    quoteTags.innerHTML = '';
                    if (quoteData.tags && quoteData.tags.length > 0) {
                        quoteData.tags.forEach(tag => {
                            const tagElement = document.createElement('span');
                            tagElement.className = 'tag';
                            tagElement.textContent = tag;
                            quoteTags.appendChild(tagElement);
                        });
                    }
                    
                    // Générer une image de prévisualisation
                    generateQuoteImage(quoteData.content, quoteData.author);
                    
                    quoteResult.classList.remove('hidden');
                    loading.classList.add('hidden');
                } catch (error) {
                    console.error('Erreur:', error);
                    loading.classList.add('hidden');
                    
                    // Utiliser une citation par défaut en cas d'erreur
                    currentQuote = {
                        content: "La vie n'est pas un problème à résoudre mais une réalité à expérimenter.",
                        author: "Søren Kierkegaard",
                        tags: ["philosophie", "vie"]
                    };
                    
                    quoteText.textContent = `"${currentQuote.content}"`;
                    quoteAuthor.textContent = `- ${currentQuote.author}`;
                    
                    // Afficher les tags
                    quoteTags.innerHTML = '';
                    currentQuote.tags.forEach(tag => {
                        const tagElement = document.createElement('span');
                        tagElement.className = 'tag';
                        tagElement.textContent = tag;
                        quoteTags.appendChild(tagElement);
                    });
                    
                    // Générer une image de prévisualisation
                    generateQuoteImage(currentQuote.content, currentQuote.author);
                    
                    quoteResult.classList.remove('hidden');
                }
            });
            
            // Trouver une vidéo d'illustration
            findVideoButton.addEventListener('click', async function() {
                if (!currentQuote) {
                    alert('Veuillez d\'abord générer une citation');
                    return;
                }
                
                // Vérifier qu'au moins une clé API est disponible
                if (!apiKeys.pexels && !apiKeys.pixabay) {
                    alert('Veuillez entrer au moins une clé API (Pexels ou Pixabay)');
                    return;
                }
                
                try {
                    loading.classList.remove('hidden');
                    findVideoButton.disabled = true;
                    
                    // Déterminer la requête de recherche
                    const searchQuery = currentQuote.tags && currentQuote.tags.length > 0 
                        ? currentQuote.tags[0] 
                        : currentQuote.content.split(' ').slice(0, 3).join(' ');
                    
                    let videoFound = false;
                    
                    // Essayer Pexels en premier si une clé est disponible
                    if (apiKeys.pexels) {
                        try {
                            const pexelsResponse = await fetch(
                                `https://api.pexels.com/videos/search?query=${encodeURIComponent(searchQuery)}&per_page=10`,
                                {
                                    headers: {
                                        'Authorization': apiKeys.pexels
                                    }
                                }
                            );
                            
                            if (pexelsResponse.ok) {
                                const pexelsData = await pexelsResponse.json();
                                if (pexelsData.videos && pexelsData.videos.length > 0) {
                                    // Choisir une vidéo aléatoire
                                    const randomVideo = pexelsData.videos[Math.floor(Math.random() * pexelsData.videos.length)];
                                    // Prendre la vidéo de la plus haute qualité disponible
                                    const videoFile = randomVideo.video_files.sort((a, b) => b.width - a.width)[0];
                                    currentVideo = videoFile.link;
                                    videoFound = true;
                                }
                            }
                        } catch (e) {
                            console.error('Erreur avec l\'API Pexels:', e);
                        }
                    }
                    
                    // Si Pexels n'a pas fonctionné, essayer Pixabay
                    if (!videoFound && apiKeys.pixabay) {
                        try {
                            const pixabayResponse = await fetch(
                                `https://pixabay.com/api/videos/?key=${apiKeys.pixabay}&q=${encodeURIComponent(searchQuery)}&per_page=10`
                            );
                            
                            if (pixabayResponse.ok) {
                                const pixabayData = await pixabayResponse.json();
                                if (pixabayData.hits && pixabayData.hits.length > 0) {
                                    // Choisir une vidéo aléatoire
                                    const randomVideo = pixabayData.hits[Math.floor(Math.random() * pixabayData.hits.length)];
                                    currentVideo = randomVideo.videos.large.url;
                                    videoFound = true;
                                }
                            }
                        } catch (e) {
                            console.error('Erreur avec l\'API Pixabay:', e);
                        }
                    }
                    
                    if (videoFound) {
                        // Afficher la vidéo
                        videoElement.src = currentVideo;
                        videoPreview.classList.remove('hidden');
                        previewContent.classList.remove('hidden');
                        noPreview.classList.add('hidden');
                    } else {
                        alert('Aucune vidéo trouvée. Essayez avec une autre citation ou vérifiez vos clés API.');
                    }
                    
                    loading.classList.add('hidden');
                    findVideoButton.disabled = false;
                } catch (error) {
                    console.error('Erreur lors de la recherche vidéo:', error);
                    loading.classList.add('hidden');
                    findVideoButton.disabled = false;
                    alert('Erreur lors de la recherche de vidéo. Veuillez réessayer.');
                }
            });
            
            // Générer une image avec la citation (simulation)
            function generateQuoteImage(text, author) {
                // En réalité, vous utiliseriez Canvas ou une API pour générer une image
                // Ici, nous utilisons une image placeholder avec la citation en alt
                quoteImage.src = 'data:image/svg+xml;utf8,<svg xmlns="http://www.w3.org/2000/svg" width="500" height="500" viewBox="0 0 500 500"><rect width="100%" height="100%" fill="%234f46e5"/><text x="50%" y="50%" dominant-baseline="middle" text-anchor="middle" font-family="Arial" font-size="24" fill="white">' + encodeURIComponent('"' + text + '" - ' + author) + '</text></svg>';
                
                previewContent.classList.remove('hidden');
                noPreview.classList.add('hidden');
            }
            
            // Préparer pour Instagram
            instagramBtn.addEventListener('click', function() {
                if (!currentQuote || !currentVideo) {
                    alert('Veuillez d\'abord générer une citation et une vidéo');
                    return;
                }
                alert('Fonctionnalité à implémenter: Export pour Instagram\nCitation: ' + currentQuote.content + '\nVidéo: ' + currentVideo);
            });
            
            // Préparer pour TikTok
            tiktokBtn.addEventListener('click', function() {
                if (!currentQuote || !currentVideo) {
                    alert('Veuillez d\'abord générer une citation et une vidéo');
                    return;
                }
                alert('Fonctionnalité à implémenter: Export pour TikTok\nCitation: ' + currentQuote.content + '\nVidéo: ' + currentVideo);
            });
        });
    </script>
